// 2. Calculator Function
function calculate() {
  const num1 = parseFloat(document.getElementById('num1').value);
  const num2 = parseFloat(document.getElementById('num2').value);
  const operator = document.getElementById('operator').value;
  
  let result;
  switch(operator) {
    case '+': result = num1 + num2; break;
    case '-': result = num1 - num2; break;
    case '*': result = num1 * num2; break;
    case '/': result = num2 !== 0 ? num1 / num2 : 'Error: Division by zero'; break;
    case '%': result = num1 % num2; break;
    case '**': result = num1 ** num2; break;
    default: result = 'Invalid operator';
  }
  
  return result;
}

// 3. Grading System
function checkGrade() {
  const score = parseInt(document.getElementById('grade').value);
  
  if (score >= 90) return 'A - Excellent!';
  else if (score >= 80) return 'B - Good!';
  else if (score >= 70) return 'C - Fair';
  else if (score >= 60) return 'D - Need Improvement';
  else return 'F - Failed';
}

// 4. Loop Examples
function demoForLoop() {
  let output = '';
  for (let i = 1; i <= 5; i++) {
    output += `Iterasi ${i}: ${i * i}
`;
  }
  return output;
}
